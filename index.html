responsive image lazy loading

This demo uses buttons to trigger the image lazy loading event. But feel free to use other event types (e.g. scroll events). 

var initLazyBtn = function (btn, placeholderId) {
    btn.onclick = function() {
        var placeholder = document.getElementById(placeholderId);
        if (placeholder) {
            Justlazy.lazyLoadImg(placeholder);
        }

        return false;
    };
};
