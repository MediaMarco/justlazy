justlazy.js

This demo uses buttons to trigger the lazy loading event. But feel free to use other event types (e.g. scroll events). 

var initLazyBtn = function (btn, placeholderId) {
    btn.onclick = function() {
        var placeholder = document.getElementById(placeholderId);
        if (placeholder) {
            Justlazy.lazyLoadImg(placeholder);
        }

        return false;
    };
};
